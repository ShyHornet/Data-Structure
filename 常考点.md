###计算叶节点数量
```c
///先序非递归
int countLeavies(BTree BT){
    int num  = 0;
    StackType ST;
		BTNode p  = NULL;
    InitStack(ST);
    ClearStack(ST);
		
    Push(ST,BT);
		while(!EmptyStack(ST)){
		
      p = Pop(ST);
			
      if (p->lchild==NULL&&p->rchild)
			{//说明是叶节点
        num++;
      }else{
			
          if(p->lchild!=NULL)
            Push(ST,p->lchild);
						
          if(p->rchild!=NULL)
            Push(ST,p->rchild);
        
      }
    
    }
  return num;

}
```
